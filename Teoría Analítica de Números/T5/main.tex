\textbf{Ejercicio 1.} Sea $f(x)$ una función definida para todo racional $x\in[0,1]$ y defina
\[
F(n)=\sum_{k=1}^{n} f\!\left(\frac{k}{n}\right),
\qquad
F_1(n)=\sum_{\substack{1\leq k\leq n\\ (k,n)=1}} f\!\left(\frac{k}{n}\right).
\]
\begin{enumerate}[(i)]
    \item Muestre que $F_1=\mu * F$ (convolución de Dirichlet).

    \begin{proof}
  Considere la suma con condición de coprimalidad, en efecto

  \begin{align*}
    F_1(n)&=\sum_{\substack{k=1\\(k,n)}}^n f\left(\frac{k}{n}\right)=\sum_{k=1}^n e((k,n))f\left(\frac{k}{n}\right)\\
    &=\sum_{k\leq n}\left(\sum_{\substack{d\mid k\\d\mid n}}\mu(d)\right)f\left(\frac{k}{n}\right)\\
    &=\sum_{d\mid n}\mu(d)\left(\sum_{\substack{k\leq n\\d\mid k}}f\left(\frac{x}{n}\right)\right) &&\text{Cambiando el orden de sumación}\\
    &=\sum_{d\mid n}\mu(d)\left(\sum_{dq\leq n}f\left(\frac{dq}{n}\right)\right)\\
    &=\sum_{d\mid n}\mu(d)\left(\sum_{q\leq n/d}f\left(\frac{q}{n/d}\right)\right)=\sum_{d\mid n}\mu(d)F\left(\frac{n}{d}\right)
  .\end{align*}

  Esto es, sencillamente que $F_1=\mu*F$, lo que se quería probar.
\end{proof}

    \item Pruebe que $\mu(n)$ es la suma de las raíces $n$-ésimas primitivas de la unidad:
    \[
    \mu(n)=\sum_{\substack{1\leq k\leq n\\(k,n)=1}} e^{2\pi i k/n}.
    \]

 \begin{proof}
      Sea $F(n)=\displaystyle\sum_{k=1}^ne^{2\pi ik/n}$, recordemos que estas son las raices del polinomomio $x^n-1$, así pues las fórmulas de Vieta nos dan que

      $$F(n)=\begin{cases}
        1 \quad n=1,\\
        0 \quad n>1
      \end{cases}$$

     obtenemos que $F(n)=e(n)=(1*\mu)(n)$, por el numeral anterior $F_1=\mu* F=\mu*e=\mu$, esto es 

     \[F_1=\sum_{\substack{1\leq k\leq n\\(k,n)=1}} e^{2\pi i k/n}=\mu(n).
    \]
    \end{proof}   
\end{enumerate}

    \item Si $\alpha>0$ y $\alpha\neq 1$, entonces
    \[
    \sum_{n\leq x}\sigma_\alpha(n)

    \zeta(\alpha+1)\frac{x^{\alpha+1}}{\alpha+1}
    +O\!\left(x^\beta\right),
    \qquad
    \beta=\max\{1,\alpha\}.
    \]

    \item Si $\alpha<0$ y $\beta=\max\{0,1+\alpha\}$, entonces para $x\geq 2$,
    \[
    \sum_{n\leq x}\sigma_\alpha(n)

    \end{cases}
    \]
\end{enumerate}
\textbf{Ejercicio 3:} Sea

$$F(s)=\sum_{n=1}^\infty \dfrac{a_n}{n^s},$$

donde $a_n$ es completamente multiplicativa y la serie converge absolutamente para $\sigma>\sigma_a$. Probar que para $\sigma>\sigma_a$ se cumple que 

$$\dfrac{F^\prime(s)}{F(s)}=-\sum_{n=1}^\infty\dfrac{a_n\Lambda(n)}{n^s}$$

\begin{lemma}
    Sean $f$ y $g$ las funciones aritmética asociadas a $F(s)$ y $G(s)$, considere $h=f*g$ y $H(s)$ su serie de Dirichlet asociada. Si $F(s)$ y $G(s)$ convergen absolutamente en $s$, entonces $H(s)$ también y además $H(s)=F(s)G(s)$ 
\end{lemma}

\begin{proof}
     Por la convergencia absoluta
     \begin{align*}
         F(s)G(s)&=\sum_{k=1}^\infty\dfrac{f(k)}{k^s}\sum_{m=1}^\infty\dfrac{g(m)}{m^s}\\
         &=\sum_{k=1}^\infty\sum_{m=1}^\infty\dfrac{f(k)g(m)}{k^sm^s}\\
         &=\sum_{n=1}^{\infty}\dfrac{1}{n^s}\sum_{km=n}f(k)g(m)\\
         &=\sum_{n=1}^\infty\dfrac{(f*g)(n)}{n^s}
     \end{align*}

     La convergencia absoluta de $H(s)$ se sigue de 

     $$\sum_{n=1}^\infty\left|\dfrac{h(n)}{n^s}\right|=\sum_{n=1}^{\infty}\dfrac{1}{|n^s|}\sum_{km=n}|f(k)||g(m)|=\sum_{k=1}^\infty\left|\dfrac{f(k)}{k^s}\right|\sum_{m=1}^\infty\left|\dfrac{g(m)}{m^s}\right|<\infty$$
\end{proof}

\begin{lemma}
    Sea $f$ una función completamente multiplicativa, entonces $(f(g*h))(n)=(fg*fh)(n)$.
\end{lemma}

\begin{proof}
Por hipótesis, $f$ es completamente multiplicativa, esto es, $f(n)=f(j)f\left(\dfrac{n}{j}\right)$, así pues
    \begin{align*}
        (f(g*h))(n)&=f(n)\sum_{j\mid n}g(j)h\left(\dfrac{n}{j}\right)\\
        &=\sum_{j\mid n}g(j)f(j)h\left(\dfrac{n}{j}\right)f\left(\dfrac{n}{j}\right)\\
        &=\sum_{j\mid n}fg(j)fh\left(\dfrac{n}{j}\right)\\
        &=(fg*fh)(n).
    \end{align*}
\end{proof}

Ahora procedemos con la prueba.

 \begin{proof}
     En efecto, para $\sigma>\sigma_a$ se cumple que

    \begin{align*}
        F^\prime(s)&=-\sum_{n=1}^\infty\dfrac{a_n\log n}{n^s}\\
        &=-\sum_{n=1}^\infty\dfrac{a_n(\Lambda*1)(n)}{n^s} &&\text{Teorema Fundamental de la Aritmética}\\
        &=-\sum_{n=1}^\infty\dfrac{a_n\Lambda(n)*a_n}{n^s}\\
        &=-\sum_{n=1}^\infty\dfrac{a_n\Lambda(n)}{n^s}\sum_{n=1}^\infty\dfrac{a_n}{n^s}=-\sum_{n=1}^\infty\dfrac{a_n\Lambda(n)}{n^s}F(s),
    \end{align*}
    donde la separación de esta convolución como un producto, está justificada por el lema anterior, ya que $F(s)$ converge absolutamente en el semiplano $\sigma>\sigma_a$, además como $\Lambda(n)\leq \log n$ y $F^\prime(s)$ converge absolutamente en $\sigma>\sigma_a$, entonces también lo hace $\displaystyle\sum_{n=1}^\infty\dfrac{a_n\Lambda(n)}{n^s}.$ De esto se sigue el resultado.
 \end{proof}


\textbf{Ejercicio 4.} Sea
\[
F(s)=\sum_{n=1}^{\infty} \frac{f(n)}{n^{s}},
\]
convergente para $\sigma>c$, donde $c$ es finito, y suponga que $f(n)\ge 0$ para todo $n\ge n_0$.

\begin{enumerate}[(a)]
    \item Demostrar que si $F(s)$ es analítica en un disco alrededor de $s=c$, entonces la serie converge en $\sigma>c-\varepsilon$ para algún $\varepsilon>0$.

    \item Concluir que si la abscisa de convergencia $\sigma_c$ es finita, entonces $F(s)$ tiene una singularidad en el punto $s=\sigma_c$ sobre el eje real.
\end{enumerate}

\begin{proof}
    Sea $a=1+c$, como $F$ es analítica en $a$, entonces por el teorema de Taylor

    \begin{align}
        F(s)=\sum_{k=0}^\infty\dfrac{F^{(k)}(a)}{k!}(s-a)^k,
    \end{align}

 además el radio de convergencia de esta serie de potencias es mayor que 1, ya que $F$ es analítica en $c$. Como vimos $\displaystyle F^\prime(s)=-\sum_{n=1}^\infty\dfrac{f(n)\log n}{n^s}$, luego derivando de forma iterada, obtenemos que

    $$F^{(k)}(a)=(-1)^k \sum_{n=1}^{\infty}f(n)(\log n)^kn^{-a}$$

podemos reescribir (1) como:

\begin{align}
F(s)=\sum_{k=0}^{\infty}\sum_{n=1}^\infty\dfrac{(a-s)^k}{k!}f(n)(\log n)^k n^{-a}
\end{align}

Nuevamente, como el radio de convergencia de (2) es mayor que 1, entonces, es convergente para algún $s\in \mathbb{R}$ tal que $s=c-\varepsilon$, con $\varepsilon>0$. En efecto $a-s=1+\varepsilon$, así pues, la serie (2) tiene términos no negativos para $n\geq n_0$, por lo tanto, el teorema de Tonelli permite cambiar el orden de sumación. Obtenemos que

$$F(c-\varepsilon)=\sum_{n=1}^\infty\dfrac{f(n)}{n^a}\sum_{k=0}^\infty\dfrac{((1+\varepsilon)\log n)^k}{k!}=\sum_{n=1}^\infty\dfrac{f(n)}{n^a}e^{(1+\varepsilon)\log n}=\sum_{n=1}^\infty \dfrac{f(n)}{n^{c-e}}.$$

Esto es, $F(s)=\displaystyle\sum_{n=1}^\infty f(n)n^{-s}$ converge para $s=c-\varepsilon$, entonces en particular, converge en el semiplano $\sigma>c-\varepsilon$.
\end{proof}

Finalmente, supongamos que $\sigma_c$ no es una singularidad de $F$, luego $F(s)$ es analítica en $s=\sigma_c$, pero por lo anterior, esto significa que $F(s)$ converge en $\sigma>c-\varepsilon$ lo que contradice que $\sigma_c$ es la abscisa de convergencia.
